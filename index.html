                    <div id="pals-result"></div>

                `;

            }

        }


        function calculateFluid() {

            const w = parseFloat(document.getElementById('weight').value);

            const temp = parseFloat(document.getElementById('temp').value);

            let isl = w < 10 ? 40 : w < 20 ? 30 : 20;

            const fever = Math.max(0, temp - 37) * 12;

            const adjusted = Math.round(isl * (1 + fever/100));

            const resultDiv = document.getElementById('fluid-result');

            resultDiv.innerHTML = `<div class="alert" style="background:#d4edda;">ISL: <strong>${adjusted} mL/kg/day</strong></div>`;

        }


        function calculatePALS() {

            const w = parseFloat(document.getElementById('pals-weight').value);

            const html = `

                Adrenaline: ${(0.01 * w).toFixed(2)} mg<br>

                Fluid Bolus: ${20 * w} mL<br>

                ETT Size (cuffed): ${(w/4 + 3.5).toFixed(1)} mm<br>

                Defibrillation: 2 J/kg → 4 J/kg

            `;

            document.getElementById('pals-result').innerHTML = `<div class="alert" style="background:#cce5ff;">${html}</div>`;

        }

        }
